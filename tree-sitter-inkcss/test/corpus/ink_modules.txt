interface Gradient {
  direction: string
  from: string
  to: string
}

function linearGradient({ direction, from, to }: Gradient) {
  return linear-gradient(=direction, =from, =to)
}

export const gradientText = (gradient: string) => {
  return {
    background: =gradient
    backgroundClip: text
    color: transparent
  }
}

export { linearGradient }

export default {
  =gradientText
} as const
